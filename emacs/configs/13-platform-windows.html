(defun m180/font-set ()
  "Font set for Windows."
  (set-face-attribute 'default nil :family "Iosevka NFM" :height 120 :weight 'light)
  (set-fontset-font t 'hangul (font-spec :name "Noto Sans KR")))

(if (daemonp)
      (add-hook 'server-after-make-frame-hook #'m180/font-set)
  (m180/font-set))
