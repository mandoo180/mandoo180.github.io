* Normal Mode Mappings

** Basic Keymaps

#+begin_src lua
-- Normal mode mappings
vim.keymap.set("n", "<leader>c", ":nohlsearch<CR>", { desc = "Clear search highlights" })

-- Y to EOL
vim.keymap.set("n", "Y", "y$", { desc = "Yank to end of line" })
#+end_src

** Navigation with Centering

#+begin_src lua
-- Center screen when jumping
vim.keymap.set("n", "n", "nzzzv", { desc = "Next search result (centered)" })
vim.keymap.set("n", "N", "Nzzzv", { desc = "Previous search result (centered)" })
vim.keymap.set("n", "<C-d>", "<C-d>zz", { desc = "Half page down (centered)" })
vim.keymap.set("n", "<C-u>", "<C-u>zz", { desc = "Half page up (centered)" })
#+end_src

** Buffer Navigation

#+begin_src lua
-- Buffer navigation
vim.keymap.set("n", "<leader>]", ":bnext<CR>", { desc = "Next buffer" })
vim.keymap.set("n", "<leader>[", ":bprevious<CR>", { desc = "Previous buffer" })
vim.keymap.set("n", "<leader>k", ":bd%<CR>", { desc = "Kill current buffer" })
#+end_src

** Window Management

#+begin_src lua
-- Splitting & Resizing
vim.keymap.set("n", "<leader>|", ":vsplit<CR>", { desc = "Split window vertically" })
vim.keymap.set("n", "<leader>_", ":split<CR>", { desc = "Split window horizontally" })
vim.keymap.set("n", "<C-Up>", ":resize +2<CR>", { desc = "Increase window height" })
vim.keymap.set("n", "<C-Down>", ":resize -2<CR>", { desc = "Decrease window height" })
vim.keymap.set("n", "<C-Left>", ":vertical resize -2<CR>", { desc = "Decrease window width" })
vim.keymap.set("n", "<C-Right>", ":vertical resize +2<CR>", { desc = "Increase window width" })
#+end_src

** Line Movement

#+begin_src lua
-- Move lines up/down
vim.keymap.set("n", "<A-j>", ":m .+1<CR>==", { desc = "Move line down" })
vim.keymap.set("n", "<A-k>", ":m .-2<CR>==", { desc = "Move line up" })
#+end_src

** File Operations

#+begin_src lua
-- Quick file navigation
-- vim.keymap.set("n", "<leader>e", ":Neotree toggle<CR>", { desc = "Open file explorer" })
vim.keymap.set("n", "<leader>ff", ":find ", { desc = "Find file" })

-- Quick config editing
vim.keymap.set("n", "<leader>rc", ":e ~/.config/nvim/init.lua<CR>", { desc = "Edit config" })

** Miscellaneous

#+begin_src lua
-- Better J behavior
vim.keymap.set("n", "J", "mzJ`z", { desc = "Join lines and keep cursor position" })

-- Escape mapping
local opts = { noremap = true, silent = true }
vim.keymap.set("n", "<C-g>", "<ESC>", opts)
#+end_src

* Visual Mode Mappings

** Paste and Delete

#+begin_src lua
-- Better paste behavior
vim.keymap.set("x", "<leader>p", '"_dP', { desc = "Paste without yanking" })

-- Delete without yanking
vim.keymap.set({ "n", "v" }, "<leader>d", '"_d', { desc = "Delete without yanking" })
#+end_src

** Line Movement in Visual Mode

#+begin_src lua
vim.keymap.set("v", "<A-j>", ":m '>+1<CR>gv=gv", { desc = "Move selection down" })
vim.keymap.set("v", "<A-k>", ":m '<-2<CR>gv=gv", { desc = "Move selection up" })
#+end_src

** Indentation

#+begin_src lua
-- Better indenting in visual mode
vim.keymap.set("v", "<", "<gv", { desc = "Indent left and reselect" })
vim.keymap.set("v", ">", ">gv", { desc = "Indent right and reselect" })
#+end_src

** Escape in Visual Mode

#+begin_src lua
vim.keymap.set("v", "<C-g>", "<ESC>", opts)
#+end_src

* Useful Functions

** Copy Full File Path

#+begin_src lua
-- ============================================================================
-- USEFUL FUNCTIONS
-- ============================================================================

-- Copy Full File-Path
vim.keymap.set("n", "<leader>pa", function()
  local path = vim.fn.expand("%:p")
  vim.fn.setreg("+", path)
  print("file:", path)
end)
#+end_src
